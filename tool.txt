#!/bin/bash

if [ $# -ne 1 ]; then
  echo "Usage: $0 <domain>"
  exit 1
fi

domain="$1"
wordlist="best-dns-wordlist.txt"

# Function to resolve a subdomain using host command with timeout
resolve_subdomain() {
  subdomain="$1"
  if timeout 1 host "$subdomain.$domain" >/dev/null 2>&1; then
    echo "$subdomain.$domain"
  fi
}

# Read wordlist line by line and resolve subdomains one by one
while read -r sub; do
  resolve_subdomain "$sub"
  sleep 5  # 5-second break between each subdomain resolution
done < "$wordlist"
